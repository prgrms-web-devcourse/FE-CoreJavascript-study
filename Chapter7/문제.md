class Expenditure {
  constructor(name, jan, feb) {
    this.name = name;
    this.jan = jan;
    this.feb = feb;
  }

  sum() {
    return this.jan + this.feb;
  }
}

class UpdateExpenditure extends Expenditure {
  constructor(name, jan, feb, mar) {
    super(name, jan, feb);
    this.mar = mar;
  }

  total() {
    return super.sum() + this.mar;
  }

  delete() {
    delete super.sum;
  }
}

let jieun = new UpdateExpenditure("jieun", 60, 70, 50);

jieun.sum(); // (1)
jieun.total(); // (2)
jieun.delete(); // (3)
